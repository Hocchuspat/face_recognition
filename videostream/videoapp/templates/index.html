args=parser.parse_args()

def highlightFace(net, frame, conf_threshold=0.7):
  # –¥–µ–ª–∞–µ–º –∫–æ–ø–∏—é —Ç–µ–∫—É—â–µ–≥–æ –∫–∞–¥—Ä–∞
  frameOpencvDnn=frame.copy()

  frameWidth=frameOpencvDnn.shape[1]
  # –ø—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –∫–∞—Ä—Ç–∏–Ω–∫—É –≤ –¥–≤–æ–∏—á–Ω—ã–π –ø–∏–∫—Å–µ–ª—å–Ω—ã–π –æ–±—ä–µ–∫—Ç
  blob=cv2.dnn.blobFromImage(frameOpencvDnn, 1.0, (300, 300), [104, 117, 123], True, False)
  # —É—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º —ç—Ç–æ—Ç –æ–±—ä–µ–∫—Ç –∫–∞–∫ –≤—Ö–æ–¥–Ω–æ–π –ø–∞—Ä–∞–º–µ—Ç—Ä –¥–ª—è –Ω–µ–π—Ä–æ—Å–µ—Ç–∏

  detections=net.forward()
  # –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è –¥–ª—è —Ä–∞–º–æ–∫ –≤–æ–∫—Ä—É–≥ –ª–∏—Ü–∞
  faceBoxes=[]
  # –ø–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ –±–ª–æ–∫–∏ –ø–æ—Å–ª–µ —Ä–∞—Å–ø–æ–∑–Ω–∞–≤–∞–Ω–∏—è
  for i in range(detections.shape[2]):
      # –ø–æ–ª—É—á–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç –≤—ã—á–∏—Å–ª–µ–Ω–∏–π –¥–ª—è –æ—á–µ—Ä–µ–¥–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
      confidence=detections[0,0,i,2]

      if confidence>conf_threshold:

          x1=int(detections[0,0,i,3]*frameWidth)

          y2=int(detections[0,0,i,6]*frameHeight)
          # –¥–æ–±–∞–≤–ª—è–µ–º –∏—Ö –≤ –æ–±—â—É—é –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é
          faceBoxes.append([x1,y1,x2,y2])

          cv2.rectangle(frameOpencvDnn, (x1,y1), (x2,y2), color, int(round(frameHeight/150)), 8)
  # –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –∫–∞–¥—Ä —Å —Ä–∞–º–∫–∞–º–∏
  return frameOpencvDnn,faceBoxes

      face=frame[max(0,faceBox[1]):
